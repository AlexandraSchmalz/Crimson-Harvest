void changeSelection(Day newSelection){
  // selecting gap day
  if(newSelection.day == 0){
    _isSelected = false;
  }
  // selecting valid day
  else{
    _selectedDay = newSelection;
    _isSelected = true;
  }
  notifyListeners();
}
